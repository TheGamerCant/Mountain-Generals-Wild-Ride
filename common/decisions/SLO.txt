SLO_deportations_to_mongolia_decision_category = {
	SLO_deportations_to_mongolia = { 

		icon = decision_eng_blackshirt_march
		
		target_trigger = {
			FROM = { 
				OR = {
					state = 43
					state = 154
					state = 973
					state = 974
					state = 155
					state = 76
					state = 45
					state = 155
				}
			}
		}

		state_target = yes
		on_map_mode = map_only

		allowed = {
			original_tag = SLO
		}
		
		ai_will_do = {
			base = 50
		}

		available = {
			FROM = {
				is_owned_by = SLO
				is_fully_controlled_by = SLO
			}
			HUN = {
				exists = yes
				any_owned_state = {
					is_core_of = HUN
					is_on_continent = asia
				}
			}
		}

		visible = {
			FROM = {
				is_owned_by = SLO
			}
		}
		
		days_remove = 60
		
		
		fire_only_once = yes

		custom_cost_trigger = {
			command_power > 24
			has_political_power > 24
		}
		custom_cost_text = decision_cost_CP_25_pp_25

		modifier = {
			political_power_cost = 0.25
		}

		complete_effect = {	
			FROM = {
				set_variable = { THIS.state_population_k_hold = THIS.state_population_k }
				
				set_variable_to_random = {
					var = THIS.random_num
					min = 0.83
					max = 0.98
				}
				
				multiply_variable = { THIS.state_population_k_hold = THIS.random_num }
				multiply_variable = { THIS.state_population_k_hold = -10 }
				
				round_variable = THIS.state_population_k_hold
			}
		}
		
		remove_effect={
			FROM = {
				for_loop_effect = {
					end = 100
					value = temp
					
					add_manpower = var:THIS.state_population_k_hold
				}
				
				random_state = {
					limit = {
						is_owned_by = HUN
						is_core_of = HUN
						is_on_continent = asia
					}
					
					#Some of you may die, but that is a sacrifice I am willing to make
					multiply_variable = { PREV.state_population_k_hold = -0.98 }
					
					for_loop_effect = {
						end = 100
						value = temp
						
						add_manpower = var:PREV.state_population_k_hold
					}
				}
				
				clear_variable = PREV.random_num
				clear_variable = PREV.state_population_k_hold
			}
		}
	}
